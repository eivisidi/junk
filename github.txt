FIRST TIME PROCESS ONLY

aptitude install git xclip
git config --global user.name "John Doe"
git config --global user.email johndoe@gmail.com
git config --global core.editor vim
git config --global merge.tool vimdiff

generate ssh key:
cd ~/.ssh
ssh-keygen -t rsa -C "johndoe@gmail.com"
xclip -sel clip < ~/.ssh/id_rsa.pub
or vim ~/.ssh/id_rsa.pub 
copy key (begins with ssh-rsa.. ends with username.com)
github.com --> account settings --> ssh keys -->
add ssh key --> paste key --> add key --> 
enter password
ssh -T git@github.com
yes

create local directory:
mkdir junk
cd junk
git init
touch filename
git add filename
git commit -m 'first commit'
git remote add origin git@github.com:username/junk.git
git push origin master
